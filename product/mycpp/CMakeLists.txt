cmake_minimum_required(VERSION 3.10)

# project name
project(MyCppProject LANGUAGES CXX)

# C++ standard
set(CMAKE_CXX_STANDARD 17)

# include directories

# set output directory for executables
set(CMAKE_RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/bin)

    src/hello.cpp
